    97→  complete: { label: 'Complete', color: 'bg-emerald-500', icon: CheckCircle2 },
    98→  failed: { label: 'Failed', color: 'bg-red-500', icon: XCircle },

   100→
   101→export default function Home() {
   102→  const [pipelineStatus, setPipelineStatus] = useState<PipelineStatus | null>(null);
   103→  const [videos, setVideos] = useState<VideoItem[]>([]);
   104→  const [config, setConfig] = useState<Config | null>(null);
   105→  const [isGenerating, setIsGenerating] = useState(false);
   106→  const [customTopic, setCustomTopic] = useState('');
   107→  const [selectedVideo, setSelectedVideo] = useState<VideoItem | null>(null);
   108→  const [showSettings, setShowSettings] = useState(false);
   109→  const [progress, setProgress] = useState(0);
   110→  const [progressLabel, setProgressLabel] = useState('');
   111→  const [logs, setLogs] = useState<string[]>([]);
   112→
   113→  const addLog = useCallback((msg: string) => {
   114→    setLogs((prev) => [...prev.slice(-50), `[${new Date().toLocaleTimeString()}] ${msg}`]);
   115→  }, []);
   116→
   117→  const fetchStatus = useCallback(async () => {
   118→    try {
   119→      const res = await fetch('/api/pipeline');
   120→      const data = await res.json();
   121→      setPipelineStatus(data);
   122→      setVideos(data.recentVideos || []);
   123→
   124→      // Derive progress from the latest active video status
   125→      const activeVideo = (data.recentVideos || []).find((v: VideoItem) =>
   126→        ['generating_script', 'generating_voiceover', 'capturing_screenshots', 'assembling'].includes(v.status)
   127→      );
   128→      if (activeVideo) {
   129→        setIsGenerating(true);
   130→        const statusProgress: Record<string, { pct: number; label: string }> = {
   131→          generating_script: { pct: 20, label: 'Writing script...' },
   132→          capturing_screenshots: { pct: 40, label: 'Capturing screenshots...' },
   133→          generating_voiceover: { pct: 60, label: 'Recording voice...' },
   134→          assembling: { pct: 85, label: 'Assembling video...' },
   135→        };
   136→        const sp = statusProgress[activeVideo.status];
   137→        if (sp) {
   138→          setProgress(sp.pct);
   139→          setProgressLabel(sp.label);
   140→        }
   141→        // Detect newly completed videos
   142→        if (activeVideo.status === 'complete') {
   143→          addLog(`✅ Video complete: ${activeVideo.title}`);
   144→        }
   145→        if (activeVideo.status === 'failed') {
   146→          addLog(`❌ Video failed: ${activeVideo.error || activeVideo.title}`);
   147→        }
   148→      } else if (isGenerating) {
   149→        // Was generating, now no active video - it finished
   150→        setIsGenerating(false);
   151→        setProgress(100);
   152→        setProgressLabel('Done!');
   153→      }
   154→    } catch (err) {
   155→      // silent
   156→    }
   157→  }, [addLog, isGenerating]);
   158→
   159→  const fetchConfig = useCallback(async () => {
   160→    try {
   161→      const res = await fetch('/api/config');
   162→      const data = await res.json();
   163→      setConfig(data);
   164→    } catch {
   165→      // ignore
   166→    }
   167→  }, []);
   168→
   169→  useEffect(() => {
   170→    const load = async () => {
   171→      await fetchStatus();
   172→      await fetchConfig();
   173→    };
   174→    load();
   175→    const interval = setInterval(() => { fetchStatus(); }, 5000);
   176→    return () => clearInterval(interval);
   177→    // eslint-disable-next-line react-hooks/exhaustive-deps
   178→  }, []);
   179→
   180→  const runPipeline = async (topic?: string) => {
   181→    setLogs([]);
   182→    addLog('🚀 Starting pipeline...');
   183→    setProgress(5);
   184→    setProgressLabel('Starting...');
   185→
   186→    try {
   187→      const res = await fetch('/api/pipeline', {
   188→        method: 'POST',
   189→        headers: { 'Content-Type': 'application/json' },
   190→        body: JSON.stringify({
   191→          topic: topic || undefined,
   192→          voiceName: config?.voiceName || 'alloy',
   193→          voiceSpeed: config?.voiceSpeed || 1.0,
   194→        }),
   195→      });
   196→
   197→      const data = await res.json();
   198→      if (data.success) {
   199→        addLog('Pipeline started, processing in background...');
   200→        setIsGenerating(true);
   201→      } else {
   202→        addLog(`❌ ${data.error}`);
   203→        setIsGenerating(false);
   204→        setProgress(0);
   205→        setProgressLabel(`Failed: ${data.error}`);
   206→      }
   207→      fetchStatus();
   208→    } catch (err: any) {
   209→      addLog(`❌ Pipeline error: ${err.message}`);
   210→      setIsGenerating(false);
   211→    }
   212→  };
   213→
   214→  const deleteVideo = async (id: string) => {
   215→    await fetch(`/api/videos?id=${id}`, { method: 'DELETE' });
   216→    fetchStatus();
   217→    setSelectedVideo(null);
   218→  };
   219→
   220→  const updateConfig = async (updates: Partial<Config>) => {
   221→    if (!config) return;
   222→    const res = await fetch('/api/config', {
   223→      method: 'PUT',
   224→      headers: { 'Content-Type': 'application/json' },
   225→      body: JSON.stringify({ ...config, ...updates }),
   226→    });
   227→    const data = await res.json();
   228→    setConfig(data);
   229→  };
   230→
   231→  const getVideoFileUrl = (video: VideoItem, type: 'video' | 'voiceover') => {
   232→    const filename = type === 'video' ? `${video.id}.mp4` : `${video.id}_voiceover.mp3`;
   233→    return `/api/serve?file=${filename}`;
   234→  };
   235→
   236→  const parseImages = (imagesJson: string): { url: string; caption?: string }[] => {
   237→    try {
   238→      return JSON.parse(imagesJson);
   239→    } catch {
   240→      return [];
   241→    }
   242→  };
   243→
   244→  return (
   245→    <div className="min-h-screen bg-[#0a0a0f] text-white">
   246→      {/* Ambient gradient background */}
   247→      <div className="fixed inset-0 pointer-events-none">
   248→        <div className="absolute top-0 left-1/4 w-96 h-96 bg-purple-500/10 rounded-full blur-3xl" />
   249→        <div className="absolute bottom-0 right-1/4 w-96 h-96 bg-cyan-500/10 rounded-full blur-3xl" />
   250→        <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-[600px] h-[600px] bg-emerald-500/5 rounded-full blur-3xl" />
   251→      </div>
   252→
   253→      <div className="relative z-10">
   254→        {/* Header */}
   255→        <header className="border-b border-white/5 backdrop-blur-xl bg-black/20">
   256→          <div className="max-w-7xl mx-auto px-4 sm:px-6 py-4 flex items-center justify-between">
   257→            <div className="flex items-center gap-3">
   258→              <div className="relative">
   259→                <div className="w-10 h-10 rounded-xl bg-gradient-to-br from-purple-500 to-cyan-500 flex items-center justify-center shadow-lg shadow-purple-500/20">
   260→                  <Zap className="w-5 h-5 text-white" />
   261→                </div>
   262→                <div className="absolute -top-1 -right-1 w-3 h-3 rounded-full bg-emerald-400 animate-pulse" />
   263→              </div>
   264→              <div>
   265→                <h1 className="text-lg font-bold tracking-tight">
   266→                  AI Slop Farm
   267→                </h1>
   268→                <p className="text-xs text-white/40">YouTube Shorts Factory</p>
   269→              </div>
   270→            </div>
   271→            <div className="flex items-center gap-3">
   272→              <Badge
   273→                variant="outline"
   274→                className={
   275→                  pipelineStatus?.status === 'running'
   276→                    ? 'border-yellow-500/50 text-yellow-400 bg-yellow-500/10'
   277→                    : 'border-emerald-500/50 text-emerald-400 bg-emerald-500/10'
   278→                }
   279→              >
   280→                {pipelineStatus?.status === 'running' ? (
   281→                  <Loader2 className="w-3 h-3 mr-1 animate-spin" />
   282→                ) : (
   283→                  <CheckCircle2 className="w-3 h-3 mr-1" />
   284→                )}
   285→                {pipelineStatus?.status === 'running' ? 'Processing' : 'Idle'}
   286→              </Badge>
   287→              <Button
   288→                variant="ghost"
   289→                size="icon"
   290→                className="text-white/60 hover:text-white hover:bg-white/10"
   291→                onClick={() => setShowSettings(true)}
   292→              >
   293→                <Settings className="w-5 h-5" />
   294→              </Button>
   295→            </div>
   296→          </div>
   297→        </header>
   298→
   299→        <main className="max-w-7xl mx-auto px-4 sm:px-6 py-6 space-y-6">
   300→          {/* Stats Row */}
   301→          <div className="grid grid-cols-2 md:grid-cols-4 gap-3">
   302→            {[
   303→              { label: 'Total Generated', value: pipelineStatus?.total || 0, icon: BarChart3, color: 'text-purple-400' },
   304→              { label: 'Processing', value: pipelineStatus?.processing || 0, icon: Loader2, color: 'text-yellow-400' },
   305→              { label: 'Completed', value: pipelineStatus?.completed || 0, icon: CheckCircle2, color: 'text-emerald-400' },
   306→              { label: 'Failed', value: pipelineStatus?.failed || 0, icon: XCircle, color: 'text-red-400' },
   307→            ].map((stat) => (
   308→              <Card key={stat.label} className="bg-white/[0.03] border-white/[0.06] backdrop-blur-sm">
   309→                <CardContent className="p-4">
   310→                  <div className="flex items-center justify-between">
   311→                    <div>
   312→                      <p className="text-xs text-white/40">{stat.label}</p>
   313→                      <p className={`text-2xl font-bold mt-1 ${stat.color}`}>{stat.value}</p>
   314→                    </div>
   315→                    <stat.icon className={`w-5 h-5 ${stat.color} opacity-50`} />
   316→                  </div>
   317→                </CardContent>
   318→              </Card>
   319→            ))}
   320→          </div>
   321→
   322→          {/* Generate Section */}
   323→          <Card className="bg-white/[0.03] border-white/[0.06] backdrop-blur-sm overflow-hidden">
   324→            <CardContent className="p-0">
   325→              {/* Progress bar at top when generating */}
   326→              <AnimatePresence>
   327→                {isGenerating && (
   328→                  <motion.div
   329→                    initial={{ height: 0, opacity: 0 }}
   330→                    animate={{ height: 'auto', opacity: 1 }}
   331→                    exit={{ height: 0, opacity: 0 }}
   332→                  >
   333→                    <div className="p-4 border-b border-white/[0.06] space-y-2">
   334→                      <div className="flex items-center justify-between">
   335→                        <div className="flex items-center gap-2">
   336→                          <Loader2 className="w-4 h-4 text-purple-400 animate-spin" />
   337→                          <span className="text-sm text-white/70">{progressLabel}</span>
   338→                        </div>
   339→                        <span className="text-sm text-white/40">{progress}%</span>
   340→                      </div>
   341→                      <Progress value={progress} className="h-1.5 bg-white/10" />
   342→                    </div>
   343→                  </motion.div>
   344→                )}
   345→              </AnimatePresence>
   346→
   347→              <div className="p-6 space-y-4">
   348→                <div className="flex flex-col sm:flex-row items-start sm:items-end gap-4">
   349→                  <div className="flex-1 space-y-2 w-full">
   350→                    <Label className="text-white/60 text-xs uppercase tracking-wider">Topic (leave blank for auto)</Label>
   351→                    <Input
   352→                      placeholder="e.g., Google just released Gemini 2.5..."
   353→                      value={customTopic}
   354→                      onChange={(e) => setCustomTopic(e.target.value)}
   355→                      className="bg-white/[0.05] border-white/[0.1] text-white placeholder:text-white/30 h-11"
   356→                      onKeyDown={(e) => e.key === 'Enter' && !isGenerating && runPipeline(customTopic)}
   357→                      disabled={isGenerating}
   358→                    />
   359→                  </div>
   360→                  <Button
   361→                    onClick={() => runPipeline(customTopic)}
   362→                    disabled={isGenerating}
   363→                    className="h-11 px-8 bg-gradient-to-r from-purple-600 to-cyan-600 hover:from-purple-500 hover:to-cyan-500 text-white font-semibold shadow-lg shadow-purple-500/20 hover:shadow-purple-500/30 transition-all"
   364→                  >
   365→                    {isGenerating ? (
   366→                      <>
   367→                        <Loader2 className="w-4 h-4 mr-2 animate-spin" />
   368→                        Generating...
   369→                      </>
   370→                    ) : (
   371→                      <>
   372→                        <Sparkles className="w-4 h-4 mr-2" />
   373→                        Generate Short
   374→                      </>
   375→                    )}
   376→                  </Button>
   377→                </div>
   378→              </div>
   379→            </CardContent>
   380→          </Card>
   381→
   382→          {/* Main Content: Videos + Logs */}
   383→          <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
   384→            {/* Video Queue */}
   385→            <div className="lg:col-span-2 space-y-4">
   386→              <div className="flex items-center justify-between">
   387→                <h2 className="text-lg font-semibold flex items-center gap-2">
   388→                  <Video className="w-5 h-5 text-purple-400" />
   389→                  Video Queue
   390→                </h2>
   391→                <Tabs defaultValue="all" className="w-auto">
   392→                  <TabsList className="bg-white/[0.05]">
   393→                    <TabsTrigger value="all" className="text-white/60 data-[state=active]:text-white data-[state=active]:bg-white/10 text-xs px-3">All</TabsTrigger>
   394→                    <TabsTrigger value="complete" className="text-white/60 data-[state=active]:text-white data-[state=active]:bg-white/10 text-xs px-3">Done</TabsTrigger>
   395→                    <TabsTrigger value="failed" className="text-white/60 data-[state=active]:text-white data-[state=active]:bg-white/10 text-xs px-3">Failed</TabsTrigger>
   396→                  </TabsList>
   397→                </Tabs>
   398→              </div>
   399→
   400→              {videos.length === 0 ? (
   401→                <Card className="bg-white/[0.03] border-white/[0.06]">
   402→                  <CardContent className="p-12 text-center">
   403→                    <div className="w-16 h-16 mx-auto mb-4 rounded-2xl bg-white/[0.05] flex items-center justify-center">
   404→                      <Film className="w-8 h-8 text-white/20" />
   405→                    </div>
   406→                    <h3 className="text-white/40 font-medium">No videos yet</h3>
   407→                    <p className="text-white/20 text-sm mt-1">Hit &quot;Generate Short&quot; to start the farm</p>
   408→                  </CardContent>
   409→                </Card>
   410→              ) : (
   411→                <ScrollArea className="max-h-[600px]">
   412→                  <div className="space-y-3 pr-4">
   413→                    <AnimatePresence>
   414→                      {videos.map((video, i) => {
   415→                        const statusCfg = STATUS_CONFIG[video.status] || STATUS_CONFIG.draft;
   416→                        const StatusIcon = statusCfg.icon;
   417→                        const images = parseImages(video.images);
   418→
   419→                        return (
   420→                          <motion.div
   421→                            key={video.id}
   422→                            initial={{ opacity: 0, y: 20 }}
   423→                            animate={{ opacity: 1, y: 0 }}
   424→                            exit={{ opacity: 0, y: -20 }}
   425→                            transition={{ delay: i * 0.05 }}
   426→                          >
   427→                            <Card
   428→                              className="bg-white/[0.03] border-white/[0.06] hover:bg-white/[0.05] hover:border-white/[0.1] transition-all cursor-pointer group"
   429→                              onClick={() => setSelectedVideo(video)}
   430→                            >
   431→                              <CardContent className="p-4">
   432→                                <div className="flex gap-4">
   433→                                  {/* Thumbnail / Status */}
   434→                                  <div className="flex-shrink-0">
   435→                                    {images.length > 0 && video.status === 'complete' ? (
   436→                                      <div className="w-20 h-28 rounded-lg overflow-hidden bg-white/[0.05] relative group/play">
   437→                                        <img
   438→                                          src={images[0].url}
   439→                                          alt=""
   440→                                          className="w-full h-full object-cover"
   441→                                          onError={(e) => {
   442→                                            (e.target as HTMLImageElement).style.display = 'none';
   443→                                          }}
   444→                                        />
   445→                                        <div className="absolute inset-0 bg-black/40 flex items-center justify-center opacity-0 group-hover/play:opacity-100 transition-opacity">
   446→                                          <div className="w-8 h-8 rounded-full bg-white/90 flex items-center justify-center">
   447→                                            <Play className="w-3.5 h-3.5 text-black ml-0.5" fill="black" />
   448→                                          </div>
   449→                                        </div>
   450→                                        {video.duration && (
   451→                                          <span className="absolute bottom-1 right-1 text-[9px] bg-black/70 text-white px-1 rounded">
   452→                                            0:{video.duration.toString().padStart(2, '0')}
   453→                                          </span>
   454→                                        )}
   455→                                      </div>
   456→                                    ) : images.length > 0 ? (
   457→                                      <div className="w-20 h-28 rounded-lg overflow-hidden bg-white/[0.05]">
   458→                                        <img
   459→                                          src={images[0].url}
   460→                                          alt=""
   461→                                          className="w-full h-full object-cover opacity-60"
   462→                                          onError={(e) => {
   463→                                            (e.target as HTMLImageElement).style.display = 'none';
   464→                                          }}
   465→                                        />
   466→                                      </div>
   467→                                    ) : (
   468→                                      <div className="w-20 h-28 rounded-lg bg-white/[0.05] flex items-center justify-center">
   469→                                        <StatusIcon className={`w-6 h-6 ${video.status === 'complete' ? 'text-emerald-400' : video.status === 'failed' ? 'text-red-400' : 'text-white/30'}`} />
   470→                                      </div>
   471→                                    )}
   472→                                  </div>
   473→
   474→                                  {/* Info */}
   475→                                  <div className="flex-1 min-w-0">
   476→                                    <div className="flex items-start justify-between gap-2">
   477→                                      <h3 className="font-medium text-sm line-clamp-1 group-hover:text-purple-400 transition-colors">
   478→                                        {video.title}
   479→                                      </h3>
   480→                                      <Badge
   481→                                        variant="outline"
   482→                                        className={`text-[10px] px-1.5 py-0 flex-shrink-0 ${
   483→                                          video.status === 'complete'
   484→                                            ? 'border-emerald-500/50 text-emerald-400'
   485→                                            : video.status === 'failed'
   486→                                            ? 'border-red-500/50 text-red-400'
   487→                                            : 'border-white/20 text-white/50'

   489→                                      >
   490→                                        {statusCfg.label}
   491→                                      </Badge>
   492→                                    </div>
   493→
   494→                                    {video.hook && (
   495→                                      <p className="text-xs text-white/40 mt-1 line-clamp-2 italic">
   496→                                        &quot;{video.hook}&quot;
   497→                                      </p>
   498→                                    )}
   499→
   500→                                    {video.sourceTitle && (
   501→                                      <p className="text-xs text-white/30 mt-1 line-clamp-1">
   502→                                        Source: {video.sourceTitle}
   503→                                      </p>
   504→                                    )}
   505→
   506→                                    <div className="flex items-center gap-3 mt-2 text-[10px] text-white/30">
   507→                                      <span className="flex items-center gap-1">
   508→                                        <Clock className="w-3 h-3" />

   510→                                      </span>
   511→                                      <span className="flex items-center gap-1">
   512→                                        <Calendar className="w-3 h-3" />
   513→                                        {new Date(video.createdAt).toLocaleDateString()}
   514→                                      </span>
